import pandas as pd
import numpy as np

  1. Import the penguin dataset https://raw.githubusercontent.com/allisonhorst/palmerpenguins/master/inst/extdata/penguins.csv

penguins = pd.read_csv('https://raw.githubusercontent.com/allisonhorst/palmerpenguins/master/inst/extdata/penguins.csv')

  1. Create a new dataframe new_dataframe by extracting the species, bill_length_mm and body_mass_g columns.

new_dataframe = penguins[['species', 'bill_length_mm','body_mass_g']]
new_dataframe.head(3)
species bill_length_mm body_mass_g
0 Adelie 39.1 3750.0
1 Adelie 39.5 3800.0
2 Adelie 40.3 3250.0

new_dataframe.loc[4]

new_dataframe.iloc[0:3, 1]
